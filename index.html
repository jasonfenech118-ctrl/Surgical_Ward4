return v_name = 'svetlana fenech'; end; $$; create or replace function public.sw4_staff_ward_id(p_staff_id uuid) returns uuid language sql security definer set search_path = public as $$ select s.home_ward_id from public.staff s where s.id = p_staff_id limit 1; $$; create or replace function public.sw4_reason_for_code(p_code text) returns text language sql immutable as $$ select case upper(coalesce(p_code,'')) when 'L' then 'annual_leave' when 'VL' then 'annual_leave' when 'SL' then 'sick_leave' when 'ML' then 'maternity_leave' when 'ST' then 'study_leave' when 'TI' then 'til_in' when 'TO' then 'til_out' when 'OT' then 'overtime' when 'O' then 'off' when 'R' then 'rest' when 'D' then 'manual_day_change' when 'N' then 'manual_night_change' else 'manual_roster_change' end; $$; create or replace function public.sw4_apply_roster_override( p_staff_id uuid, p_shift_date date, p_shift_code text, p_notes text default null ) returns void language plpgsql security definer set search_path = public as $$ declare v_ward_id uuid; exception when undefined_function then return; when others then return; end; end; $$; -- --------------------------------------------------------- -- Ensure Leave / TIL / Overtime request table exists -- --------------------------------------------------------- create table if not exists public.sw4_leave_requests ( id uuid primary key default gen_random_uuid(), staff_id uuid not null references public.staff(id), request_date date not null, request_code text not null, is_hourly boolean not null default false, start_time time without time zone, hours numeric(5,2), status text not null default 'pending', staff_note text, approver_note text, approved_by_staff_id uuid references public.staff(id), approved_at timestamptz, created_at timestamptz not null default now(), updated_at timestamptz not null default now() ); alter table public.sw4_leave_requests add column if not exists is_hourly boolean not null default false; alter table public.sw4_leave_requests add column if not exists start_time time without time zone; alter table public.sw4_leave_requests add column if not exists hours numeric(5,2); alter table public.sw4_leave_requests add column if not exists status text not null default 'pending'; alter table public.sw4_leave_requests add column if not exists staff_note text; create index if not exists sw4_leave_requests_date_idx on public.sw4_leave_requests(request_date); -- --------------------------------------------------------- -- Ensure Change of Duty table exists -- --------------------------------------------------------- create table if not exists public.sw4_change_of_duty ( id uuid primary key default gen_random_uuid(), staff_id uuid not null references public.staff(id), cod_type text not null, off_date date not null, worked_date date not null, notes text, status text not null default 'pending', created_by_staff_id uuid references public.staff(id), approved_by_staff_id uuid references public.staff(id), approved_at timestamptz, approver_note text, created_at timestamptz not null default now(), updated_at timestamptz not null default now() ); alter table public.sw4_change_of_duty add column if not exists created_by_staff_id uuid references public.staff(id); alter table public.sw4_change_of_duty add column if not exists approved_by_staff_id uuid references public.staff(id); alter table public.sw4_change_of_duty add column if not exists approved_at timestamptz; alter table public.sw4_change_of_duty add column if not exists approver_note text; alter table public.sw4_change_of_duty add column if not exists created_at timestamptz not null default now(); alter table public.sw4_change_of_duty add column if not exists updated_at timestamptz not null default now(); alter table public.sw4_change_of_duty enable row level security; create index if not exists sw4_cod_staff_dates_idx on public.sw4_change_of_duty(staff_id, off_date, worked_date); create function public.submit_sw4_leave_request( p_request_date date, p_request_code text, p_is_hourly boolean default false, p_start_time time without time zone default null, p_hours numeric default null, p_staff_note text default null ) returns uuid language plpgsql security definer set search_path = public as $$ declare v_staff_id uuid; v_code text; v_id uuid; v_status text; v_auto_approved boolean; begin v_staff_id := public.sw4_current_staff_id(); if v_staff_id is null then raise exception 'No linked staff profile was found for this login.'; end if; v_code := upper(trim(coalesce(p_request_code,''))); if v_code in ('AL','ANNUAL','ANNUAL_LEAVE','LEAVE','VACATION','VACATION_LEAVE') then v_code := 'L'; end if; if v_code in ('S','SICK','SICK_LEAVE') then v_code := 'SL'; end if; if v_code in ('MATERNITY','MATERNITY_LEAVE') then v_code := 'ML'; end if; if v_code in ('STUDY','STUDY_LEAVE') then v_code := 'ST'; end if; if v_code in ('TILIN','TIL_IN') then v_code := 'TI'; end if; if v_code in ('TILOUT','TIL_OUT') then v_code := 'TO'; end if; if v_code in ('OVERTIME') then v_code := 'OT'; end if; if v_code not in ('L','SL','ML','ST','TI','TO','OT') then raise exception 'Invalid request type: %', p_request_code; end if; if coalesce(p_is_hourly,false) and (p_hours is null or p_hours <= 0 or p_hours > 24) then raise exception 'Hourly requests must have hours between 1 and 24.'; end if; v_auto_approved := public.sw4_is_full_control(); v_status := case when v_auto_approved then 'approved' else 'pending' end; insert into public.sw4_leave_requests( staff_id, request_date, request_code, is_hourly, start_time, hours, status, staff_note, approver_note, approved_by_staff_id, approved_at ) values ( v_staff_id, p_request_date, v_code, coalesce(p_is_hourly,false), case when coalesce(p_is_hourly,false) then coalesce(p_start_time,'07:00'::time) else null end, case when coalesce(p_is_hourly,false) then p_hours else null end, v_status, p_staff_note, case when v_auto_approved then 'Automatically approved for Svetlana/full-control user.' else null end, case when v_auto_approved then v_staff_id else null end, case when v_auto_approved then now() else null end ) returning id into v_id; if v_auto_approved and not coalesce(p_is_hourly,false) then perform public.sw4_apply_roster_override( v_staff_id, p_request_date, case when v_code = 'VL' then 'L' else v_code end, 'Automatically approved SW4 request for Svetlana/full-control user: ' || v_code ); end if; return v_id; end; $$; -- --------------------------------------------------------- -- Change of Duty function -- New rule confirmed: Svetlana/full-control records are approved immediately. -- Normal staff records remain pending for Svetlana approval. -- --------------------------------------------------------- drop function if exists public.create_sw4_change_of_duty(uuid,text,date,date,text); create function public.create_sw4_change_of_duty( p_staff_id uuid, p_cod_type text, p_off_date date, p_worked_date date, p_notes text default null ) returns uuid language plpgsql security definer set search_path = public as $$ declare v_id uuid; v_actor uuid; v_type text; v_status text; begin v_actor := public.sw4_current_staff_id(); if v_actor is null then raise exception 'No linked staff profile was found for this login.'; end if; v_type := lower(trim(coalesce(p_cod_type,''))); if v_type not in ('normal','public_holiday','sunday') then raise exception 'Invalid Change of Duty type.'; end if; if not public.sw4_is_full_control() and p_staff_id <> v_actor then raise exception 'Staff can only record Change of Duty for themselves.'; end if; v_status := case when public.sw4_is_full_control() then 'approved' else 'pending' end; insert into public.sw4_change_of_duty( staff_id, cod_type, off_date, worked_date, notes, status, created_by_staff_id, approved_by_staff_id, approved_at, approver_note ) values ( p_staff_id, v_type, p_off_date, p_worked_date, p_notes, v_status, v_actor, case when v_status = 'approved' then v_actor else null end, case when v_status = 'approved' then now() else null end, case when v_status = 'approved' then 'Automatically approved for Svetlana/full-control user.' else null end ) returning id into v_id; if v_status = 'approved' then perform public.sw4_apply_roster_override(p_staff_id, p_off_date, 'O', 'COD off vice ' || to_char(p_worked_date,'DD/MM/YYYY')); perform public.sw4_apply_roster_override(p_staff_id, p_worked_date, 'D', 'COD on vice ' || to_char(p_off_date,'DD/MM/YYYY')); end if; return v_id; end; $$; -- --------------------------------------------------------- -- Convert any existing pending Svetlana leave/TIL/OT records to approved. -- Also applies roster overlays for full-day records. -- --------------------------------------------------------- do $$ declare v_svetlana_staff_id uuid; r record; begin select s.id into v_svetlana_staff_id from public.staff s where lower(coalesce(s.email,'')) = 'svetlana.fenech@gov.mt' or lower(coalesce(s.full_name,'')) = 'svetlana fenech' order by case when lower(coalesce(s.email,'')) = 'svetlana.fenech@gov.mt' then 0 else 1 end, s.created_at nulls last limit 1; if v_svetlana_staff_id is not null and to_regclass('public.sw4_leave_requests') is not null then update public.sw4_leave_requests set status = 'approved', approved_by_staff_id = coalesce(approved_by_staff_id, v_svetlana_staff_id), approved_at = coalesce(approved_at, now()), approver_note = coalesce(approver_note, 'Automatically approved for Svetlana/full-control user.'), updated_at = now() where staff_id = v_svetlana_staff_id and lower(coalesce(status,'')) = 'pending'; for r in select id, request_date, upper(coalesce(request_code,'')) as request_code from public.sw4_leave_requests where staff_id = v_svetlana_staff_id and lower(coalesce(status,'')) = 'approved' and coalesce(is_hourly,false) = false loop perform public.sw4_apply_roster_override( v_svetlana_staff_id, r.request_date, case when r.request_code = 'VL' then 'L' else r.request_code end, 'Automatically approved SW4 request for Svetlana/full-control user: ' || r.request_code ); end loop; end if; end $$; -- --------------------------------------------------------- -- Convert any existing pending Svetlana Change of Duty records to approved. -- Also applies COD roster markings. -- --------------------------------------------------------- do $$ declare v_svetlana_staff_id uuid; r record; begin select s.id into v_svetlana_staff_id from public.staff s where lower(coalesce(s.email,'')) = 'svetlana.fenech@gov.mt' or lower(coalesce(s.full_name,'')) = 'svetlana fenech' order by case when lower(coalesce(s.email,'')) = 'svetlana.fenech@gov.mt' then 0 else 1 end, s.created_at nulls last limit 1; if v_svetlana_staff_id is not null and to_regclass('public.sw4_change_of_duty') is not null then update public.sw4_change_of_duty set status = 'approved', approved_by_staff_id = coalesce(approved_by_staff_id, v_svetlana_staff_id), approved_at = coalesce(approved_at, now()), approver_note = coalesce(approver_note, 'Automatically approved for Svetlana/full-control user.'), updated_at = now() where staff_id = v_svetlana_staff_id and lower(coalesce(status,'')) = 'pending'; for r in select id, off_date, worked_date from public.sw4_change_of_duty where staff_id = v_svetlana_staff_id and lower(coalesce(status,'')) = 'approved' loop perform public.sw4_apply_roster_override(v_svetlana_staff_id, r.off_date, 'O', 'COD off vice ' || to_char(r.worked_date,'DD/MM/YYYY')); perform public.sw4_apply_roster_override(v_svetlana_staff_id, r.worked_date, 'D', 'COD on vice ' || to_char(r.off_date,'DD/MM/YYYY')); end loop; end if; end $$; -- --------------------------------------------------------- -- Grants -- --------------------------------------------------------- grant execute on function public.sw4_current_staff_id() to authenticated; grant execute on function public.sw4_is_full_control() to authenticated; grant execute on function public.sw4_staff_ward_id(uuid) to authenticated; grant execute on function public.sw4_reason_for_code(text) to authenticated;
